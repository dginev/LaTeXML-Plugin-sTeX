\begin{modsig}[creators=miko]{relation}
\symdef{sseteq}[2]{#1\subset #2}
\symdef{pairs}[2]{#1\times #2}
\end{modsig}

\begin{modnl}[creators=miko]{relation}{en}
\begin{definition}[id=relation.def]
  $\sseteq{R}{\pairs{A}B}$ is a (binary) \defi{relation} between $A$ and $B$.
  If $A=B$ then $R$ is called a \defii{relation}{on} $A$.
\end{definition}
\end{modnl}
